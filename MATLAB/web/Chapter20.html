% Example of the d-simplex, first in the euclidean plane (p=2, d=2)
% A picture reveals that the radius of the inscribed circle is 1/(2+sqrt(2))
d = 2; % ambient dimension
p = 2; % index of the \ell_p-norm
q = p/(p-1); % conjugate index
A = [ones(1,d); -eye(d)];
b = [1; zeros(d,1)];
row_norms = zeros(d+1,1);
for i = 1:d+1
row_norms(i) = norm(A(i,:),q);
end
cvx_begin quiet
variable center(d)
variable radius
maximize radius
subject to
A*center + radius*row_norms <= b
cvx_end
fprintf(['In the euclidean plane, the computed inscribed radius is %.3f,', ...
' which indeed equals 1/(2+sqrt(2))=%.3f'], radius, 1/(2+sqrt(2)))

Hausdorff distance between two bounded polytopes

Finding the Hausdorff distance, relatively to the -norm, between the polytopes and amounts to solving linear programs, where K and are the numbers of vertices of 𝒞 and of , respectively. Denoting these vertices by and , the linear programs are:
% Example of the d-simplex and shifted version of it
d = 13;
A = [ones(1,d); -eye(d)];
b = [1; zeros(d,1)];
V = [zeros(d,1), eye(d)]; %the columns of V are vertices of the simplex
shift = 2*rand(d,1)/d;
AA = A;
bb = b + A*shift;
VV = shift + V; %the columns of VV are vertices of shifted simplex
left_terms = zeros(1,d+1);
right_terms = zeros(1,d+1);
for k=1:d+1
cvx_begin quiet
variable xx(d)
variable c(d)
minimize sum(c)
subject to
AA*xx <= bb;
V(:,k)-xx <= c;
xx-V(:,k) <= c;
cvx_end
left_terms(k) = cvx_optval;
cvx_begin quiet
variable x(d)
variable c(d)
minimize sum(c)
subject to
A*x <= b;
VV(:,k)-x <= c;
x-VV(:,k) <= c;
cvx_end
right_terms(k) = cvx_optval;
end
Haus_dist = max(max(left_terms),max(right_terms))
Haus_dist = 1.1244
